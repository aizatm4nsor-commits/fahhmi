                    <h2 class="font-bold text-emerald-900 mb-2">Growth Function f(x)</h2>
                    <code class="block text-sm bg-white p-3 rounded border border-emerald-200">
                        f(x) = L / (1 + e^(-k(x - x₀)))
                    </code>
                    <p class="text-xs text-emerald-800 mt-2">L=100cm (Max height), k=0.08, x₀=50 (Inflection point)</p>
                </div>

                <div class="bg-blue-50 p-6 rounded-2xl border border-blue-100">
                    <h2 class="font-bold text-blue-900 mb-2">Growth Rate f'(x)</h2>
                    <code class="block text-sm bg-white p-3 rounded border border-blue-200">
                        f'(x) = (L * k * e^(-k(x - x₀))) / (1 + e^(-k(x - x₀)))²
                    </code>
                    <p class="text-xs text-blue-800 mt-2">Calculates cm/day (Speed of elongation)</p>
                </div>

        <div id="inspector" class="mt-8 bg-white p-6 rounded-2xl shadow-md border border-indigo-100 hidden">

            <div class="grid md:grid-cols-2 gap-4 mt-4">
                <div class="p-4 bg-slate-50 rounded-lg">
                    <p class="text-sm text-slate-500 uppercase">Instantaneous Growth Rate</p>
                    <p id="rateVal" class="text-2xl font-bold text-indigo-600">0.00 cm/day</p>
                </div>
                <div class="p-4 bg-slate-50 rounded-lg">
                    <p class="text-sm text-slate-500 uppercase">Layman Explanation</p>

    <script>
        const ctx = document.getElementById('growthChart').getContext('2d');
        const L = 100, k = 0.08, x0 = 50;

        // Generate data points
        const labels = Array.from({length: 120}, (_, i) => i);
        const data = labels.map(x => L / (1 + Math.exp(-k * (x - x0))));
        const derivatives = labels.map(x => {
            const exp = Math.exp(-k * (x - x0));
            return (L * k * exp) / Math.pow(1 + exp, 2);
        });

        const myChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Paddy Height (cm)',
                    data: data,
                    borderColor: '#059669',
                    tension: 0.4,
                    pointRadius: 0
                }]
            },
            options: {
                responsive: true,
                onClick: (event, elements) => {
                    if (elements.length > 0) {
                        const index = elements[0].index;
                        showAnalysis(index);
                    }
                }
            }
        });

        function showAnalysis(day) {
            const rate = derivatives[day].toFixed(3);
            const inspector = document.getElementById('inspector');
            inspector.classList.remove('hidden');
            document.getElementById('dayVal').innerText = day;
            document.getElementById('rateVal').innerText = `${rate} cm/day`;
            
            let explanation = "";
            if (day < 30) explanation = "Initial establishment phase. Growth is slow as the plant focuses on root development.";
            else if (day < 70) explanation = "Peak vegetative growth. The plant is elongating rapidly—optimal time for nitrogen application.";
            else explanation = "Maturity phase. Growth is plateauing as the plant shifts energy toward grain filling.";
            
            document.getElementById('explanationVal').innerText = explanation;
        }
    </script>
